  // Helper: Calculate opponent scoring run in recent events
  function calculateOpponentRun(recentEvents: StatEvent[]) {
    let run = 0;
    for (let i = recentEvents.length - 1; i >= 0; i--) {
      const e = recentEvents[i];
      if (e.opponentEvent && (e.eventType.includes('made') || e.eventType === 'points')) {
        run += e.value || (e.eventType === 'three_made' ? 3 : e.eventType === 'ft_made' ? 1 : 2);
      } else if (!e.opponentEvent && (e.eventType.includes('made') || e.eventType === 'points')) {
        break; // Our team scored, break the run
      }
    }
    return run;
  }

  // Helper: Find opponent hot hand (most points in recent events)
  function findHotOpponent(recentEvents: StatEvent[]) {
    const oppPoints: Record<string, number> = {};
    for (const e of recentEvents) {
      if (e.opponentEvent && (e.eventType.includes('made') || e.eventType === 'points')) {
        oppPoints[e.playerName] = (oppPoints[e.playerName] || 0) + (e.value || (e.eventType === 'three_made' ? 3 : e.eventType === 'ft_made' ? 1 : 2));
      }
    }
    let maxName = null, maxPoints = 0;
    for (const name in oppPoints) {
      if (oppPoints[name] > maxPoints) {
        maxPoints = oppPoints[name];
        maxName = name;
      }
    }
    return maxPoints >= 6 ? { name: maxName, points: maxPoints } : null;
  }

    if (foulTroublePlayers.length > 0) {
      recommendations.push(`Watch foul trouble: ${foulTroublePlayers.map(p => p.name).join(', ')}`);
    }

    // Rebounding
    if (teamStats.totalRebounds < opponentStats.totalRebounds) {
      recommendations.push('Crash the boards harder—opponent is out-rebounding you.');
    }

    // Turnovers
    if (teamStats.totalTurnovers > opponentStats.totalTurnovers) {
      recommendations.push('Take care of the ball—limit turnovers.');
    }

    // Shot selection
    if (teamStats.fgPercentage < 40) {
      recommendations.push('Work for higher percentage shots—attack the rim.');
    }

    // Bench usage
    const tiredPlayers = players.filter(p => p.minutesPlayed > 20);
    if (tiredPlayers.length > 0) {
      recommendations.push(`Get fresh legs in: ${tiredPlayers.map(p => p.name).join(', ')}`);
    }

    // Opponent hot hand
    const hotOpponent = findHotOpponent(recentEvents);
    if (hotOpponent) {
      recommendations.push(`Deny the ball to ${hotOpponent.name}—they are heating up (${hotOpponent.points} pts).`);
    }

    // Turnovers in recent events
    if (recentEvents.some(e => e.eventType === 'turnover')) {
      recommendations.push('Take care of the ball - no more turnovers');
    }

    return recommendations;
  }

      projectedFinal: Math.round(((playersInWindow.reduce((sum, p) => sum + p.points, 0) + gameState.opponentScore) / (120 / 60)) * (settings.quarterDuration * settings.totalQuarters)),
      fgPercentage: 0, // placeholder
    };
    teamStats.fgPercentage = teamStats.fgAttempted > 0 ? Math.round((teamStats.fgMade / teamStats.fgAttempted) * 100) : 0;

    // Opponent run/hot hand (last 2 min)
    const opponentStats = {
      totalRebounds: recentEvents.filter(e => e.opponentEvent && (e.eventType === 'defensive_rebound' || e.eventType === 'offensive_rebound')).length,
      totalTurnovers: recentEvents.filter(e => e.opponentEvent && e.eventType === 'turnover').length,
    };
    const recentOpponentRun = calculateOpponentRun(recentEvents);
    const hotOpponent = findHotOpponent(recentEvents);

    // Recommendations (reuse halftime logic, but pass windowed stats)
    const recommendations = generateHalftimeRecommendations(
      teamStats,
      playersInWindow,
      gameState,
      opponentStats,
      { momentum, pace: teamStats.pace, hotOpponent }
    );

    return {
      recentEvents,
      momentum,
      keyPlayer: topScorer, // or mostEfficient
      recommendations,
      teamStats,
      topScorer,
      topRebounder,
      topAssister,
      mostEfficient,
      recentOpponentRun,
      hotOpponent,

    opponentStats: any,
    halftimeData: any
  ) => {
    const recommendations: string[] = []

    // Momentum analysis

      recommendations.push('Look for transition opportunities and fast breaks')

    if (teamStats.threePercentage < 30) {
      recommendations.push('Consider fewer three-point attempts - focus on high-percentage shots')

    if (teamStats.assistToTurnoverRatio < 1) {
      recommendations.push('Improve ball movement and reduce turnovers')

    if (teamStats.totalTurnovers > 8) {
      recommendations.push('Take care of the ball - too many turnovers')

    if (teamStats.totalRebounds < opponentStats.totalRebounds) {
      recommendations.push('Crash the boards harder - opponent is out-rebounding you')

    // Foul trouble
    const foulTroublePlayers = players.filter(p => p.fouls >= 3)
    if (foulTroublePlayers.length > 0) {
      recommendations.push(`Manage foul trouble: ${foulTroublePlayers.map(p => p.name).join(', ')}`)

    // Bench usage
    const tiredPlayers = players.filter(p => p.minutesPlayed > 15)
    if (tiredPlayers.length > 0) {
      recommendations.push(`Get fresh legs in: ${tiredPlayers.map(p => p.name).join(', ')}`)

    // Opponent analysis
    if (halftimeData.hotOpponent) {
      recommendations.push(`Deny the ball to ${halftimeData.hotOpponent.name} - they are heating up`)

    // Lead/deficit analysis
    const scoreDiff = gameState.homeScore - gameState.opponentScore
    if (scoreDiff > 10) {
      recommendations.push('Maintain the lead - don\'t let up on defense')
    } else if (scoreDiff < -10) {
      recommendations.push('Need to pick up the pace - focus on scoring and defense')
    } else {
      recommendations.push('Close game - every possession matters')

      recommendations.push('High-paced game - manage energy and substitutions')
    } else if (halftimeData.pace < 60) {
      recommendations.push('Slow game - look for transition opportunities')
    }

          break;

          updated.rebounds += 1;
          break;
        case 'offensive_rebound':
          updated.offensiveRebounds += 1;
          updated.rebounds += 1;
          break;
        case 'defensive_rebound':
          updated.defensiveRebounds += 1;
          updated.rebounds += 1;
          break;
        case 'assist':
          updated.assists += 1;
          break;
        case 'steal':
          updated.steals += 1;
          updated.plusMinus += 2;
          break;
        case 'block':
          updated.blocks += 1;
          break;
        case 'foul':
          updated.fouls += 1;
          break;
        case 'turnover':
          updated.turnovers += 1;
          updated.plusMinus -= 2;
          break;
        case 'charge_taken':
          updated.chargesTaken += 1;
          updated.plusMinus += 2;
          break;
        case 'deflection':
          updated.deflections += 1;
          break;
        case 'fg_attempt':
          updated.fgAttempted += 1;
          break;
        case 'fg_made':
          updated.fgMade += 1;
          updated.points += 2;
          updated.plusMinus += 2;
          break;
        case 'three_attempt':
          updated.threeAttempted += 1;
          break;
        case 'three_made':
          updated.threeMade += 1;
          updated.points += 3;
          updated.plusMinus += 3;
          break;
        case 'ft_attempt':
          updated.ftAttempted += 1;
          break;
        case 'ft_made':
          updated.ftMade += 1;
          updated.points += 1;
          updated.plusMinus += 1;
          break;
      }
      return updated;
    }));

          homeScore: prev.homeScore + points

      type: 'stat',
      timestamp: Date.now(),
      data: { playerId, eventType, value, isOpponent },

                        <Text strong>Most Efficient:</Text> #{timeoutData.mostEfficient.player.number} {timeoutData.mostEfficient.player.name} (+{timeoutData.mostEfficient.efficiency})
                      </div>
                    </div>
                  </Card>
                </Col>
                <Col span={12}>
                  <Card title="📊 Team Performance" size="small" className={style.insightCard}>
                    <Row gutter={16}>
                      <Col span={12}>
                        <Statistic title="Shooting Efficiency" value={`${timeoutData.teamStats.fgPercentage}%`} />
                      </Col>
                      <Col span={12}>
                        <Statistic title="Rebound Rate" value={`${timeoutData.teamStats.totalRebounds}`} />
                      </Col>
                      <Col span={12}>
                        <Statistic title="A/T Ratio" value={timeoutData.teamStats.assistToTurnoverRatio.toFixed(2)} />
                      </Col>
                      <Col span={12}>
                        <Statistic title="Points (2min)" value={timeoutData.teamStats.totalPoints} />
                      </Col>
                    </Row>
                  </Card>
                </Col>
                <Col span={12}>
                  <Card title="🔥 Opponent Run & Hot Hand" size="small" className={style.insightCard}>
                    <div style={{ color: 'white' }}>
                      <div style={{ marginBottom: 8 }}>
                        <Text strong>Opponent Run:</Text> {timeoutData.recentOpponentRun}
                      </div>
                      {timeoutData.hotOpponent && (
                        <div>
                          <Text strong>Hot Opponent:</Text> {timeoutData.hotOpponent.name} ({timeoutData.hotOpponent.points} pts)
                        </div>
                      )}
                    </div>
                  </Card>
                </Col>
                <Col span={12}>
                  <Card title="🎯 Game Pace & Projections" size="small" className={style.insightCard}>
                    <Row gutter={16}>
                      <Col span={12}>
                        <Statistic title="Current Pace" value={`${timeoutData.pace} pts/game`} />
                      </Col>
                      <Col span={12}>
                        <Statistic title="Projected Final" value={`${timeoutData.projectedFinal} pts`} />
                      </Col>
                      <Col span={12}>
                        <Statistic title="Lead/Deficit" value={timeoutData.lead} />
                      </Col>
                    </Row>
                  </Card>
                </Col>
                <Col span={24}>
                  <Card title="💡 Strategic Recommendations" size="small" className={style.recommendations}>
                    <ul>
                      {timeoutData.recommendations.map((rec, index) => (
                        <li key={index}>{rec}</li>
                      ))}
                    </ul>
                  </Card>
                </Col>
              </Row>
            </Card>
          </Col>
        )}

        {/* Main Content Section */}
        <Col span={24}>
          {activeTab === 'tracking' && tabItems.find(tab => tab.key === 'tracking')?.children}
          {activeTab === 'analytics' && tabItems.find(tab => tab.key === 'analytics')?.children}
          {activeTab === 'lineups' && tabItems.find(tab => tab.key === 'lineups')?.children}
        </Col>
      </Row>

      {/* Export Modal */}
      <Modal
        title="Export Game Data"
        open={showExportModal}
        onCancel={() => setShowExportModal(false)}
        footer={null}
      >
        <Space direction="vertical" style={{ width: '100%' }}>
          <Button icon={<DownloadOutlined />} onClick={() => exportGameData('json')} block>
            Export as JSON
          </Button>
          <Button icon={<DownloadOutlined />} onClick={() => exportGameData('csv')} block>
            Export as CSV
          </Button>
        </Space>
      </Modal>

      {/* Settings Modal */}
      <Modal
        title="Game Settings"
        open={showSettingsModal}
        onCancel={() => setShowSettingsModal(false)}
        width={800}
        footer={[
          <Button key="reset" onClick={resetSettings}>
            Reset to Defaults
          </Button>,
          <Button key="export" onClick={exportSettings}>
            Export Settings
          </Button>,
          <Button key="cancel" onClick={() => setShowSettingsModal(false)}>
            Cancel
          </Button>,
          <Button key="save" type="primary" onClick={() => setShowSettingsModal(false)}>
            Save Settings
          </Button>
        ]}
      >
        <Tabs defaultActiveKey="game" size="small">
          <TabPane tab="Game Configuration" key="game">
            <Row gutter={[16, 16]}>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Quarter Duration (minutes):</Text>
                  <Input
                    type="number"
                    value={settings.quarterDuration}
                    onChange={(e) => saveSettings({ quarterDuration: parseInt(e.target.value) || 10 })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Total Quarters:</Text>
                  <Select
                    value={settings.totalQuarters}
                    onChange={(value) => saveSettings({ totalQuarters: value })}
                    style={{ width: '100%', marginTop: 8 }}
                  >
                    <Option value={4}>4 Quarters</Option>
                    <Option value={2}>2 Halves</Option>
                    <Option value={6}>6 Quarters (Youth)</Option>
                  </Select>
                </div>
              </Col>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Timeouts per Team:</Text>
                  <Input
                    type="number"
                    value={settings.timeoutCount}
                    onChange={(e) => saveSettings({ timeoutCount: parseInt(e.target.value) || 4 })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Shot Clock (seconds):</Text>
                  <Input
                    type="number"
                    value={settings.shotClock}
                    onChange={(e) => saveSettings({ shotClock: parseInt(e.target.value) || 30 })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
            </Row>
          </TabPane>

          <TabPane tab="Workflow" key="workflow">
            <Row gutter={[16, 16]}>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Workflow Mode:</Text>
                  <Select
                    value={settings.workflowMode}
                    onChange={(value) => saveSettings({ workflowMode: value })}
                    style={{ width: '100%', marginTop: 8 }}
                  >
                    <Option value="player-first">Player First (Select player, then action)</Option>
                    <Option value="action-first">Action First (Select action, then player)</Option>
                  </Select>
                </div>
              </Col>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Auto-pause on Timeout:</Text>
                  <Switch
                    checked={settings.autoPauseOnTimeout}
                    onChange={(checked) => saveSettings({ autoPauseOnTimeout: checked })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Auto-pause on Quarter End:</Text>
                  <Switch
                    checked={settings.autoPauseOnQuarterEnd}
                    onChange={(checked) => saveSettings({ autoPauseOnQuarterEnd: checked })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Show Confirmations:</Text>
                  <Switch
                    checked={settings.showConfirmations}
                    onChange={(checked) => saveSettings({ showConfirmations: checked })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
            </Row>
          </TabPane>

          <TabPane tab="Display" key="display">
            <Row gutter={[16, 16]}>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Show Player Numbers:</Text>
                  <Switch
                    checked={settings.showPlayerNumbers}
                    onChange={(checked) => saveSettings({ showPlayerNumbers: checked })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Show Positions:</Text>
                  <Switch
                    checked={settings.showPositions}
                    onChange={(checked) => saveSettings({ showPositions: checked })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Show Efficiency Ratings:</Text>
                  <Switch
                    checked={settings.showEfficiencyRatings}
                    onChange={(checked) => saveSettings({ showEfficiencyRatings: checked })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Compact Mode:</Text>
                  <Switch
                    checked={settings.compactMode}
                    onChange={(checked) => saveSettings({ compactMode: checked })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
            </Row>
          </TabPane>

          <TabPane tab="Export" key="export">
            <Row gutter={[16, 16]}>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Auto Export:</Text>
                  <Switch
                    checked={settings.autoExport}
                    onChange={(checked) => saveSettings({ autoExport: checked })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Export Format:</Text>
                  <Select
                    value={settings.exportFormat}
                    onChange={(value) => saveSettings({ exportFormat: value })}
                    style={{ width: '100%', marginTop: 8 }}
                  >
                    <Option value="json">JSON</Option>
                    <Option value="csv">CSV</Option>
                    <Option value="pdf">PDF</Option>
                  </Select>
                </div>
              </Col>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Export Interval (minutes):</Text>
                  <Input
                    type="number"
                    value={settings.exportInterval}
                    onChange={(e) => saveSettings({ exportInterval: parseInt(e.target.value) || 5 })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Include Player Stats:</Text>
                  <Switch
                    checked={settings.includePlayerStats}
                    onChange={(checked) => saveSettings({ includePlayerStats: checked })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Include Team Stats:</Text>
                  <Switch
                    checked={settings.includeTeamStats}
                    onChange={(checked) => saveSettings({ includeTeamStats: checked })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Include Lineup Data:</Text>
                  <Switch
                    checked={settings.includeLineupData}
                    onChange={(checked) => saveSettings({ includeLineupData: checked })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
            </Row>
          </TabPane>

          <TabPane tab="Analytics" key="analytics">
            <Row gutter={[16, 16]}>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Show Advanced Stats:</Text>
                  <Switch
                    checked={settings.showAdvancedStats}
                    onChange={(checked) => saveSettings({ showAdvancedStats: checked })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Show Projections:</Text>
                  <Switch
                    checked={settings.showProjections}
                    onChange={(checked) => saveSettings({ showProjections: checked })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Show Trends:</Text>
                  <Switch
                    checked={settings.showTrends}
                    onChange={(checked) => saveSettings({ showTrends: checked })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Highlight Top Performers:</Text>
                  <Switch
                    checked={settings.highlightTopPerformers}
                    onChange={(checked) => saveSettings({ highlightTopPerformers: checked })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Efficiency Threshold:</Text>
                  <Input
                    type="number"
                    value={settings.efficiencyThreshold}
                    onChange={(e) => saveSettings({ efficiencyThreshold: parseInt(e.target.value) || 15 })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
            </Row>
          </TabPane>

          <TabPane tab="Notifications" key="notifications">
            <Row gutter={[16, 16]}>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Halftime Reminder:</Text>
                  <Switch
                    checked={settings.halftimeReminder}
                    onChange={(checked) => saveSettings({ halftimeReminder: checked })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Timeout Reminder:</Text>
                  <Switch
                    checked={settings.timeoutReminder}
                    onChange={(checked) => saveSettings({ timeoutReminder: checked })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Foul Trouble Alert:</Text>
                  <Switch
                    checked={settings.foulTroubleAlert}
                    onChange={(checked) => saveSettings({ foulTroubleAlert: checked })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Show Strategic Recommendations:</Text>
                  <Switch
                    checked={settings.showRecommendations}
                    onChange={(checked) => saveSettings({ showRecommendations: checked })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
              <Col span={12}>
                <div className={style.settingGroup}>
                  <Text strong>Show Quick Actions:</Text>
                  <Switch
                    checked={settings.showQuickActions}
                    onChange={(checked) => saveSettings({ showQuickActions: checked })}
                    style={{ marginTop: 8 }}
                  />
                </div>
              </Col>
            </Row>
          </TabPane>
        </Tabs>
      </Modal>

      {/* Lineup Builder Modal */}
      <Modal
        title="Create New Lineup"
        open={showLineupBuilder}
        onCancel={() => setShowLineupBuilder(false)}
        footer={[
          <Button key="cancel" onClick={() => setShowLineupBuilder(false)}>
            Cancel
          </Button>,
          <Button 
            key="create" 
            type="primary" 
            onClick={createLineup}
            disabled={selectedLineupPlayers.length !== 5}
          >
            Create Lineup ({selectedLineupPlayers.length}/5)
          </Button>
        ]}
      >
        <div>
          <div style={{ marginBottom: 16 }}>
            <Text strong>Select 5 Players:</Text>
            <Input
              placeholder="Lineup name (optional)"
              value={lineupName}
              onChange={(e) => setLineupName(e.target.value)}
              style={{ marginTop: 8 }}
            />
          </div>
          <div className={style.playerSelection}>
            {players.map(player => (
              <div
                key={player.id}
                className={`${style.selectablePlayer} ${selectedLineupPlayers.includes(player.id) ? style.selected : ''}`}
                onClick={() => {
                  if (selectedLineupPlayers.includes(player.id)) {
                    setSelectedLineupPlayers(prev => prev.filter(id => id !== player.id))
                  } else if (selectedLineupPlayers.length < 5) {
                    setSelectedLineupPlayers(prev => [...prev, player.id])
                  }
                }}
              >
                <div>
                  <Text strong>#{player.number} {player.name}</Text>
                  <br />
                  <Text type="secondary">{player.position}</Text>
                </div>
                <div>
                  <Badge count={player.points} style={{ backgroundColor: '#52c41a' }} />
                  <Badge count={player.plusMinus} style={{ backgroundColor: player.plusMinus >= 0 ? '#52c41a' : '#f5222d' }} />
                </div>
              </div>
            ))}
          </div>
        </div>
      </Modal>

      {/* DEV-ONLY: Quick Substitution Modal */}
      <Modal
        title="Quick Substitution"
        open={showQuickSubModal}
        onCancel={() => {
          setShowQuickSubModal(false)
          setSubstitutionPlayerIn(null)
          setSubstitutionPlayerOut(null)
        }}
        footer={[
          <Button key="cancel" onClick={() => {
            setShowQuickSubModal(false)
            setSubstitutionPlayerIn(null)
            setSubstitutionPlayerOut(null)
          }}>
            Cancel
          </Button>,
          <Button 
            key="substitute" 
            type="primary" 
            onClick={() => {
              if (substitutionPlayerIn && substitutionPlayerOut) {
                handleQuickSubstitution(substitutionPlayerIn, substitutionPlayerOut)
              }
            }}
            disabled={!substitutionPlayerIn || !substitutionPlayerOut}
            style={{
              backgroundColor: '#2563eb',
              borderColor: '#2563eb'
            }}
          >
            Confirm Substitution
          </Button>
        ]}
        width={600}
        styles={{
          content: {
            backgroundColor: '#17375c',
            color: '#f5f7fa'
          },
          header: {
            backgroundColor: '#17375c',
            color: '#f5f7fa'
          },
          body: {
            backgroundColor: '#17375c',
            color: '#f5f7fa'
          }
        }}
      >
        <div style={{ color: '#f5f7fa' }}>
          {/* DEV-ONLY: Player Out Selection */}
          <div style={{ 
            background: '#1e3a5c', 
            padding: 16, 
            borderRadius: 8, 
            marginBottom: 16,
            border: '1px solid #334155'
          }}>
            <Text strong style={{ color: '#f5f7fa', marginBottom: 8, display: 'block', fontSize: '1rem' }}>
              Player Out:
            </Text>
            {substitutionPlayerOut ? (
              <div style={{ 
                background: '#17375c', 
                padding: 12, 
                borderRadius: 6, 
                marginBottom: 16,
                border: '1px solid #2563eb'
              }}>
                <Text strong style={{ color: '#f5f7fa', fontSize: '1rem' }}>
                  #{substitutionPlayerOut.number} {substitutionPlayerOut.name}
                </Text>
                <br />
                <Text style={{ color: '#cbd5e1', fontSize: '0.9rem' }}>
                  {substitutionPlayerOut.points}PTS • {substitutionPlayerOut.plusMinus}+/- • {substitutionPlayerOut.position}
                </Text>
              </div>
            ) : (
              <Select
                placeholder="Select player to sub out"
                style={{ width: '100%', marginBottom: 16 }}
                onChange={(value) => {
                  const player = players.find(p => p.id === value)
                  if (player) setSubstitutionPlayerOut(player)
                }}
                dropdownStyle={{ backgroundColor: '#1e293b' }}
              >
                {getCourtPlayers().map(player => (
                  <Option key={player.id} value={player.id}>
                    #{player.number} {player.name} ({player.position})
                  </Option>
                ))}
              </Select>
            )}
          </div>

          {/* DEV-ONLY: Substitution Flow */}
          {substitutionPlayerOut && (
            <div style={{ 
              display: 'flex', 
              alignItems: 'center', 
              justifyContent: 'space-between', 
              margin: '16px 0',
              padding: '16px',
              background: '#1e3a5c',
              borderRadius: 8,
              border: '1px solid #334155'
            }}>
              <div style={{ 
                flex: 1, 
                textAlign: 'center', 
                padding: 12, 
                borderRadius: 6, 
                margin: '0 8px',
                background: 'rgba(239, 68, 68, 0.2)',
                border: '1px solid rgba(239, 68, 68, 0.5)'
              }}>
                <Text strong style={{ color: '#f5f7fa', fontSize: '1.1rem' }}>
                  #{substitutionPlayerOut.number}
                </Text>
                <br />
                <Text style={{ color: '#cbd5e1', fontSize: '0.9rem' }}>
                  {substitutionPlayerOut.name}
                </Text>
              </div>
              <div style={{ 
                fontSize: '1.5rem', 
                color: '#94a3b8', 
                margin: '0 16px',
                fontWeight: 'bold'
              }}>
                →
              </div>
              <div style={{ 
                flex: 1, 
                textAlign: 'center', 
                padding: 12, 
                borderRadius: 6, 
                margin: '0 8px',
                background: 'rgba(34, 197, 94, 0.2)',
                border: '1px solid rgba(34, 197, 94, 0.5)'
              }}>
                <Text strong style={{ color: '#f5f7fa', fontSize: '1.1rem' }}>
                  {substitutionPlayerIn ? `#${substitutionPlayerIn.number}` : 'Select'}
                </Text>
                <br />
                <Text style={{ color: '#cbd5e1', fontSize: '0.9rem' }}>
                  {substitutionPlayerIn ? substitutionPlayerIn.name : 'Player'}
                </Text>
              </div>
            </div>
          )}

          {/* DEV-ONLY: Player In Selection */}
          {substitutionPlayerOut && (
            <div style={{ 
              background: '#1e3a5c', 
              padding: 16, 
              borderRadius: 8, 
              marginBottom: 16,
              border: '1px solid #334155'
            }}>
              <Text strong style={{ color: '#f5f7fa', marginBottom: 8, display: 'block', fontSize: '1rem' }}>
                Player In:
              </Text>
              {substitutionPlayerIn ? (
                <div style={{ 
                  background: '#17375c', 
                  padding: 12, 
                  borderRadius: 6, 
                  marginBottom: 16,
                  border: '1px solid #2563eb'
                }}>
                  <Text strong style={{ color: '#f5f7fa', fontSize: '1rem' }}>
                    #{substitutionPlayerIn.number} {substitutionPlayerIn.name}
                  </Text>
                  <br />
                  <Text style={{ color: '#cbd5e1', fontSize: '0.9rem' }}>
                    {substitutionPlayerIn.points}PTS • {substitutionPlayerIn.plusMinus}+/- • {substitutionPlayerIn.position}
                  </Text>
                </div>
              ) : (
                <Select
                  placeholder="Select player to sub in"
                  style={{ width: '100%', marginBottom: 16 }}
                  onChange={(value) => {
                    const player = players.find(p => p.id === value)
                    if (player) setSubstitutionPlayerIn(player)
                  }}
                  dropdownStyle={{ backgroundColor: '#1e293b' }}
                >
                  {getAvailablePlayers().map(player => (
                    <Option key={player.id} value={player.id}>
                      #{player.number} {player.name} ({player.position})
                    </Option>
                  ))}
                </Select>
              )}
            </div>
          )}

          {/* DEV-ONLY: Quick substitution buttons for common scenarios */}
          {substitutionPlayerOut && !substitutionPlayerIn && (
            <div style={{ marginTop: 16 }}>
              <Text strong style={{ color: '#f5f7fa', marginBottom: 8, display: 'block', fontSize: '1rem' }}>
                Quick Substitutions:
              </Text>
              <div style={{ display: 'grid', gridTemplateColumns: 'repeat(auto-fit, minmax(150px, 1fr))', gap: 8 }}>
                {getAvailablePlayers().slice(0, 6).map(player => (
                  <Button
                    key={player.id}
                    size="small"
                    onClick={() => setSubstitutionPlayerIn(player)}
                    style={{ 
                      background: '#2563eb',
                      borderColor: '#2563eb',
                      color: '#f5f7fa',
                      fontWeight: 600
                    }}
                  >
                    #{player.number} {player.name}
                  </Button>
                ))}
              </div>
            </div>
          )}

          {/* DEV-ONLY: Substitution history */}
          {quickSubHistory.length > 0 && (
            <div style={{ marginTop: 16 }}>
              <Text strong style={{ color: '#f5f7fa', marginBottom: 8, display: 'block', fontSize: '1rem' }}>
                Recent Substitutions:
              </Text>
              <div style={{ maxHeight: 100, overflowY: 'auto' }}>
                {quickSubHistory.slice(0, 3).map((sub, index) => (
                  <div key={index} style={{ 
                    background: '#1e3a5c', 
                    padding: '8px 12px', 
                    borderRadius: 6, 
                    marginBottom: 6,
                    fontSize: '0.9rem',
                    border: '1px solid #334155'
                  }}>
                    <Text style={{ color: '#f5f7fa' }}>
                      #{sub.playerOut.number} → #{sub.playerIn.number} 
                      <Text style={{ color: '#94a3b8', marginLeft: 8 }}>
                        ({new Date(sub.timestamp).toLocaleTimeString()})
                      </Text>
                    </Text>
                  </div>
                ))}
              </div>
            </div>
          )}
        </div>
      </Modal>
    </div>
  )
}

export default Statistics
